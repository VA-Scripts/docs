    ```lua server.lua
    RegisterNetEvent('mitscript:lukOpkald', function(callId)
        local src = source
    
        -- Uden denne linje kan enhver spiller lukke ethvert opkald.
        if not exports.va_policemdt:IsMdtAuthed(src) then return end
    
        exports.va_policemdt:CloseCall(callId)
    end)
    ```

    `IsPolice`, `IsMdtAuthed` og `HasMdtPermission` findes præcis til det formål.
